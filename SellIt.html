Project Sell It

This assignment was about designing a “tower building” game. The story we had was about mineworkers from Diamond & Co. They were captured by their boss in a pit. The mineworkers had to make 2 groups and work their way out. The first group getting out wins. This game was supposed to be a competitive game between 2 teams, each building their own tower. They had to find resources of which some were more rare then others. Each team had to find a balance between building, reinforce, collect and attack.

They key part of the project was to design a game and not to develop one. Nevertheless, we have developed a number of game mechanics to strengthen the visualization. 


void AimShoot(){
    Vector3 velocityVector = transform.forward * initialVelocity;
    LineRenderer.SetVertexCount((int)(maxTime / timeResolution));
    int index = 0;
    Vector3 currentPosition = transform.position;

    if (shoot){
        LineRenderer.enabled = true;
        landingPointDisplay.SetActive(true);
       
        if (landingPointInstance != null){
            landingPointInstance.SetActive(true);
        }
        
        if (Input.GetKeyDown(KeyCode.L)){
            objectToSpawn.SetActive(true);
            Instantiate(objectToSpawn, spawnPos.transform.position, transform.rotation);
            landingPointDisplay.SetActive(false);
            landingPointInstance.SetActive(false);
            LineRenderer.enabled = false;
            shoot = false;
        }
        
        for (float t = 0.0f; t < maxTime; t += timeResolution){
            LineRenderer.SetPosition(index, currentPosition);
            RaycastHit hit;

            if (Physics.Raycast(currentPosition, velocityVector, out hit, velocityVector.magnitude * timeResolution, layerMask)){
                LineRenderer.SetVertexCount(index + 2);
                LineRenderer.SetPosition(index + 1, hit.point);
                
                if (landingPointDisplay != null){
                    if (landingPointInstance != null){
                        if (hit.transform.gameObject.name.Contains("Building") || hit.transform.gameObject.name.Contains("Cube")){
                            landingPointDisplay.SetActive(true);
                            landingPointInstance.transform.position = hit.point;
                        }
                    }
                    else if (hit.transform.gameObject.name.Contains("Building")){
                        landingPointInstance = Instantiate(landingPointDisplay, hit.point, Quaternion.identity) as GameObject;
                        landingPointDisplay.SetActive(true);
                    }
                }
                break;
            }
            else{
                if (landingPointInstance != null){
                    landingPointDisplay.SetActive(false);
                }
            }

            currentPosition += velocityVector * timeResolution;
            velocityVector += Physics.gravity * timeResolution;
            index++;
        }
    }
}
